<script>

    const csvUrl = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vTQDOlnsORiDWKfJVWLEgF5UraDT77KsKwSXn4vj10PWk17q29VpK1uDGJriLYY4_KcfDTxoqpWx-wG/pub?output=csv';

    let rawData = [];

    let currentCat = '全部';

    let currentTag = '全部';


    // 智慧隱藏邏輯

    let lastScrollTop = 0;

    const navbar = document.getElementById('navbar');

    window.addEventListener('scroll', function() {

        let scrollTop = window.pageYOffset || document.documentElement.scrollTop;

        if (scrollTop > lastScrollTop && scrollTop > 50) {

            navbar.classList.add('nav-up');

        } else {

            navbar.classList.remove('nav-up');

        }

        lastScrollTop = scrollTop <= 0 ? 0 : scrollTop;

    }, false);


    function fixImageUrl(url) {

        if (!url) return '';

        if (url.includes('drive.google.com')) {

            let fileId = (url.includes('/d/')) ? url.split('/d/')[1].split('/')[0] : url.split('id=')[1].split('&')[0];

            return `https://drive.google.com/thumbnail?id=${fileId}&sz=w1000`;

        }

        return url;

    }


    function init() {

        Papa.parse(csvUrl, {

            download: true, header: true,

            complete: function(results) {

                rawData = results.data.filter(row => row.Category && row.Coordinates);

                renderCatFilters();

                filterLogic();

            }

        });

    }


    function renderCatFilters() {

        const cats = [...new Set(rawData.map(i => i.Category))];

        let html = `<button class="btn active" onclick="selectCat('全部', this)">全部區域</button>`;

        cats.forEach(c => html += `<button class="btn" onclick="selectCat('${c}', this)">${c}</button>`);

        document.getElementById('cat-group').innerHTML = html;

    }


    function selectCat(cat, btn) {

        currentCat = cat; currentTag = '全部';

        document.querySelectorAll('#cat-group .btn').forEach(b => b.classList.remove('active'));

        btn.classList.add('active');

        const subArea = document.getElementById('sub-filter-area');

        if (cat === '全部') {

            subArea.style.display = 'none';

        } else {

            subArea.style.display = 'block';

            renderTagFilters(cat);

        }

        filterLogic();

    }


    function renderTagFilters(cat) {

        const filteredByCat = rawData.filter(i => i.Category === cat);

        let allTags = [];

        filteredByCat.forEach(item => {

            if (item.Tags) allTags.push(...item.Tags.split(',').map(t => t.trim()));

        });

        const uniqueTags = [...new Set(allTags)].filter(t => t);

        let html = `<button class="btn btn-sub active" onclick="selectTag('全部', this)">全部類型</button>`;

        uniqueTags.forEach(t => html += `<button class="btn btn-sub" onclick="selectTag('${t}', this)">${t}</button>`);

        document.getElementById('tag-group').innerHTML = html;

    }


    function selectTag(tag, btn) {

        currentTag = tag;

        document.querySelectorAll('#tag-group .btn').forEach(b => b.classList.remove('active'));

        btn.classList.add('active');

        filterLogic();

    }


    function filterLogic() {

        let data = rawData;

        if (currentCat !== '全部') data = data.filter(i => i.Category === currentCat);

        if (currentTag !== '全部') data = data.filter(i => i.Tags && i.Tags.split(',').map(t => t.trim()).includes(currentTag));

        display(data);

    }


    function display(data) {

        const container = document.getElementById('grid');

        container.innerHTML = data.map(item => {

            const tagHtml = item.Tags ? item.Tags.split(',').map(t => `<span class="tag">${t.trim()}</span>`).join('') : '';

            return `

                <div class="card">

                    <img src="${fixImageUrl(item.ImageURL)}" onerror="this.src='https://placehold.co/400x500?text=圖片載入中'">

                    <div class="card-body">

                        <div class="tag-list">${tagHtml}</div>

                        <div class="coord-display">${item.Coordinates}</div>

                        <button class="copy-btn" onclick="copyText('${item.Coordinates}')">複製座標</button>

                    </div>

                </div>

            `;

        }).join('');

    }


    function copyText(t) {

        navigator.clipboard.writeText(t).then(() => {

            const s = document.getElementById("toast");

            s.className = "show";

            setTimeout(() => s.className = "", 2500);

        });

    }


    window.onload = init;
